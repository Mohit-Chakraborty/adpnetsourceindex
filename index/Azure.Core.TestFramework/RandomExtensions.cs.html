using System;
 
namespace Azure.Core.TestFramework
{
    public static class RandomExtensions
    {
        public static Guid NewGuid(this Random random)
        {
            var bytes = new byte[16];
            random.NextBytes(bytes);
            return new Guid(bytes);
        }
    }
}
