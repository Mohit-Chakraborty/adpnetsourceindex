    {
        /// <summary>
        /// Provides a method to convert ManifestWrapper to V2Manifest
        /// </summary>
        public static explicit operator V2Manifest(ManifestWrapper v)
        {
            var manifest = new V2Manifest
            {
                Layers = v.Layers,
                SchemaVersion = v.SchemaVersion,
                Config = v.Config,
