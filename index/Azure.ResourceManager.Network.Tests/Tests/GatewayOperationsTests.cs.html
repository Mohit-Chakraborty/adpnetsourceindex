                Location = location,
                Sku = new VirtualNetworkGatewaySku()

                {
                    new VirtualNetworkGatewayIPConfiguration()

                        Name = ipConfigName,
                        PrivateIPAllocationMethod = IPAllocationMethod.Dynamic,

                            Id = nic1publicIp.Id
