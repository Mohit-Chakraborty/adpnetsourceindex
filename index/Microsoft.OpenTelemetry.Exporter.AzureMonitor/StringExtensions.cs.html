using System;
 
namespace Microsoft.OpenTelemetry.Exporter.AzureMonitor
{
    internal static class StringExtensions
    {
        public static string Trim(this string message, int lastIndex)
        {
            return message.Substring(0, lastIndex);
        }
    }
}
