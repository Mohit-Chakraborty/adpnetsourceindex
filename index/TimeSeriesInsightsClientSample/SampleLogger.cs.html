using System;
 
namespace Azure.Iot.TimeSeriesInsights.Samples
{
    internal static class SampleLogger
    {
        internal static void FatalError(string message)
        {
            Console.ForegroundColor = ConsoleColor.Red;
            Console.WriteLine(message);
            Console.ResetColor();
            Environment.Exit(0);
        }
    }
}
